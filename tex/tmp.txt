'use strict';

$(function () {
  $('#start').on('click', function() {
      $('#area').css('display', 'block');
          $('#start').css('display', 'none');
            
});

  $('#close').on('click', function() {
      $('#area').css('display', 'none');
          $('#start').css('display', 'block');
            
});

});


@charset "UTF-8";

#area {
  display: none;
  
}

#start {
  cursor: pointer;
  
}

#close {
  cursor: pointer;
  
}
